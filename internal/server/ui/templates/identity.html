Examples

# OIDC Discovery
curl http://localhost:{{.Port}}/.well-known/openid-configuration

# Get a token (client_credentials)
curl -X POST http://localhost:{{.Port}}/token \
  -u sidequest-client:sidequest-secret \
  -d 'grant_type=client_credentials'

# Introspect a token
curl -X POST http://localhost:{{.Port}}/introspect \
  -d 'token=YOUR_TOKEN_HERE'

# Userinfo
curl -H 'Authorization: Bearer YOUR_TOKEN' \
  http://localhost:{{.Port}}/userinfo
